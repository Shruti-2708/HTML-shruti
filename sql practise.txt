Q7: Get student names and their course names from students and courses tables.
  SELECT students.name, courses.course_name
  from students
  inner join courses
  ON students.course_id = courses.course_id;

Q 13 .Show course name with total students

Select c.course_name, COUNT(s.student_id)
From courses c
left join students s
on  c.course_id = s.course_id
group by  c.course_name;

Q24 Find total students in each city (only cities with at least 2 students)

SELECT city, COUNT(*)
FROM students
GROUP BY city
HAVING COUNT(*) >= 2;

Q 25 Find maximum age from each city
SELECT city, MAX(age)

Q 26 Find students whose course is Computer Science
SELECT s.name, s.age
FROM students s
JOIN courses c
ON s.course_id = c.course_id

Q 27 Find total number of students per course (course name wise)
SELECT c.course_name, COUNT(s.student_id)
FROM courses c
LEFT JOIN students s
ON c.course_id = s.course_id
GROUP BY c.course_name;

Q 28 Find cities where average student age is above 20
SELECT city, AVG(age) 
FROM students
GROUP BY city
HAVING AVG(age) > 20;
