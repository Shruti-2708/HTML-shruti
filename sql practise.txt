Q 1: Display only name and age from the students table.
 select name,age
 from students;

Q 2: Fetch students whose age is greater than 18.
  select * from students
  where age > 18;

Q 3: Display all students ordered by age in descending order.
 select * from students
 order by age DESC;

Q 4: Fetch students whose age is above 18 and city is 'Pune'.
  select * from students 
  where age >18 AND city= 'Pune';

Q 5: Find the total number of students.
  select count(*) from students;

Q6: Find number of students in each city.
  select city, count (*)
  from students
  group by city:

Q7: Get student names and their course names from students and courses tables.
  SELECT students.name, courses.course_name
  from students
  inner join courses
  ON students.course_id = courses.course_id;

Q 8. Display students from Mumbai
   select * FROM students

Q 9 Find students whose name starts with ‘S’
  select * FROM students
  where  name LIKE 'S%';

Q 10 Show unique cities

Q 11. Find students between age 18 and 21

Q 12. Count students in each course

   Select course_id, COUNT(*)
   from students
   group by course_id;

Q 13 .Show course name with total students

Select c.course_name, COUNT(s.student_id)
From courses c
left join students s
on  c.course_id = s.course_id
group by  c.course_name;

Q.15 Find youngest student

Q. 16 Students enrolled in 'Data Science'
SELECT s.name, s.city
FROM students s
JOIN courses c
ON s.course_id = c.course_id
WHERE c.course_name = 'Data Science';
